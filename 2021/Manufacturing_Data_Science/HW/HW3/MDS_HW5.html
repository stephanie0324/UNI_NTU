def fit_model(train,valid,lrate):
    # define model
    cnn_ = Sequential([ 
    #### layer 1
    Conv2D(32, 3, activation='relu', padding='same', strides=2,input_shape=img_size+(1,)),
    MaxPooling2D(pool_size=2, strides=2),
    
    #### layer 2
    Conv2D(64, 3, activation='relu', padding='same', strides=2),
    MaxPooling2D(pool_size=2, strides=2),
    
    ##### flatten
    Flatten(),
    Dense(128, activation='relu'),
    Dense(1, activation='sigmoid') 
    ])
    
    #### compile
    cnn_.compile(optimizer=Adam(learning_rate=lrate),loss='binary_crossentropy',metrics=['accuracy'])
    
    # fit model
    history = cnn_.fit(train ,validation_data=valid,epochs=15,verbose=0)

    # plot learning curves
    plt.plot(history.history['accuracy'], label='Accuracy')
    plt.plot(history.history['val_accuracy'], label='Validation Accuracy')
    pyplot.title('lrate='+str(lrate), pad=-50)
